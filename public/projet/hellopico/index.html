int i = 0;

void setup() {

  pinMode(25, OUTPUT);
  Serial.begin(9600);

}

void loop() {

  Serial.println(i);

  analogWrite(25, 50); 
  Serial.println(" Allume");
  delay(500);

  digitalWrite(25, 0); 
  Serial.println(" Eteint");
  delay(1000);
